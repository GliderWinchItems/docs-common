// Master Controller state machine  definitions 
// Lower nibble reserved for sub-states
#define MC_SAFE      (0 << 4)
#define MC_PREP      (1 << 4)
#define MC_ARMED     (2 << 4)
#define MC_GRNDRTN   (3 << 4)
#define MC_RAMP      (4 << 4)
#define MC_CLIMB     (5 << 4)
#define MC_RECOVERY  (6 << 4)
#define MC_RETRIEVE  (7 << 4)
#define MC_ABORT     (8 << 4)
#define MC_STOP      (9 << 4)
